# 553p~

```
# SQL Injection

# rule Header 룰 하나만 잘 만들어두면 복붙해서 사용하면 됨
alert tcp $EXTERNAL_NET any -> $HOME_NET 80 (msg:"SQL Injection-union"; content:"union"; nocase; http_uri; sid:3000001; rev:1;)

alert tcp $EXTERNAL_NET any -> $HOME_NET 80 (msg:"SQL Injection-or"; content:"or"; nocase; http_uri; sid:3000002; rev:1;)

alert tcp $EXTERNAL_NET any -> $HOME_NET 80 (msg:"SQL Injection-and"; content:"and"; nocase; http_uri; sid:3000003; rev:1;)
```
